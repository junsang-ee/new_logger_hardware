Table logger {
  id varchar [primary key]
  measurement_range varchar [note: "FK (logger)N : 1(logger_measurement_range)"]
  logger_group varchar [note: "FK (logger)N : 1(logger_group)"]
  created_by varchar [note: "FK (logger)N : 1(user)"]
  status varchar [note: "WAITING, MOVEMENT, ARRIVAL"]
  serial_number int
  name varchar
  mac_address varchar
  connection_at datetime
  battery int
  created_at datetime
}

Table logger_measurement_range {
  id varchar [primary key]
  temp_range varchar [note: "(logger_measurement_range)N : 1(temperature_range)"]
  humidity_range varchar [note: "(logger_measurement_range)N : 1(humidity_range)"]
  default_single_custom_measurement_range varchar [note: "(logger_measurement_range)N : 1(default_single_custom_measurement_range)"]
  created_at datetime
}

Table temperature_range {
  id varchar [primary key]
  type varchar
  lowest_temperature int
  highest_temperature int
  custom_type varchar [note: "customType enum"]
  created_at datetime
}

Table humidity_range {
  id varchar [primary key]
  lowest_humidify double
  highest_humidify double
  custom_type varchar [note: "customType enum"]
  created_at datetime
}

Table default_single_measurement {
  id varchar [primary key]
  measurement_period int
  measurement_latency int
  crash_range int
  created_at datetime
}

Table logger_group {
  id varchar [primary key]
  created_by varchar [note: "(logger_group)N : 1(user)"]
  name varchar
  created_at datetime
}

Table user {
  id varchar [primary key]
  login_info varchar [note: "(user)1 : 1(login_info)"]
  company varchar [note: "(user)1 : 1(company)"]
  app_push varchar [note: "(user)1 : 1(app_push)"]
  registration_code varchar
  profile_image varchar
  phone_number varchar
  password varchar
  name varchar
  email varchar
  created_at datetime
}

Table user_login_info {
  id varchar [primary key]
  temp_password varchar
  login_fail_count int
  updated_at datetime
  created_at datetime
}

Table app_push {
  id varchar [primary key]
  night_reception varchar
  event_occurs varchar
  created_at datetime
}

Table company {
  id varchar [primary key]
  zip_code int
  representative_number varchar
  name varchar
  detail_address varchar
  department_name varchar
  business_number varchar
  address varchar
  created_at datetime
}

Table client {
  id varchar [primary key]
  zip_code varchar
  name varchar
  address varchar
  created_at datetime
}

Table shipping {
  id varchar [primary key]
  order_info varchar [note: "FK (shipping)1 : 1(order_info)"]
  status varchar
  arrived_at datetime
  created_at datetime
}

Table logger_status_history {
  id varchar [primary key]
  order_info varchar [note: "FK (logger_status_history)N : 1(order_info)"]
  temperature double
  humidity double
  cumulative_crash int
  created_at datetime
}

Table measurement_break_away_count {
  id varchar [primary key]
  order_info varchar [note: "FK (measurement_break_away_count)1 : 1(order_info)"]
  temperature_break_away_count int
  humidity_break_away_count int
  created_at datetime
}

Table order_info {
  id varchar [primary key]
  order_product varchar [note: "FK (order_info)1 : 1(order_product)"]
  logger varchar [note: "FK (order_info)N : 1(logger)"]
  client varchar [note: "FK (order_info)N : 1(client)"]
  order_number int
  created_at datetime
}

Table order_product {
  id varchar [primary key]
  product varchar [note: "FK (order_product)N : 1(product)"]
  order_product_count int
}

Table product {
  id varchar [primary key]
  name varchar
  cost bigint
  created_at datetime
}
