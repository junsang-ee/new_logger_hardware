Table logger {
  id varchar [primary key]
  logger_status varchar [note: "FK (logger)1 : 1(logger_status)"]
  logger_group varchar [note: "FK (logger)N : 1(logger_group)"]
  created_by varchar [note: "FK (logger)N : 1(user)"]
  status int [note: "WAIT, MOVE, ARRIVE"]
  serial_number int
  name varchar
  mac_address varchar
  connection_at datetime
  battery int
  created_at datetime
}

Table logger_group {
  id String [primary key]
  logger_status varchar
  created_at datetime
  connect_count int
  grouping_type int
  name varchar
}

Table user {
  id varchar [primary key]
  notification varchar [note: "1:N (logger)"]
  profile_image varchar
  password varchar
  name varchar
  email varchar
  created_at datetime
}

Table client {
  id varchar [primary key]
  zip_code varchar
  name varchar
  address varchar
  created_at datetime
}

Table shipping {
  id varchar [primary key]
  order_info varchar [note: "FK (shipping)1 : 1(order_info)"]
  status varchar
  arrived_at datetime
  created_at datetime
}

Table logger_status_history {
  id varchar [primary key]
  order_info varchar [note: "FK (logger_status_history)N : 1(order_info)"]
  temperature double
  humidity double
  cumulative_crash int
  created_at datetime
}

Table measurement_break_away_count {
  id varchar [primary key]
  order_info varchar [note: "FK (measurement_break_away_count)1 : 1(order_info)"]
  temperature_break_away_count int
  humidity_break_away_count int
  created_at datetime
}

Table order_info {
  id varchar [primary key]
  order_product varchar [note: "FK (order_info)1 : 1(order_product)"]
  logger varchar [note: "FK (order_info)N : 1(logger)"]
  client varchar [note: "FK (order_info)N : 1(client)"]
  order_number int
  created_at datetime
}

// logger
Ref: logger.logger_group > logger_group.id
Ref: logger.created_by > user.id

// orderInfo
Ref: order_info.logger > logger.id

// shipping
Ref: shipping.order_info - order_info.id

// logger_status_history
Ref: logger_status_history.order_info > order_info.id

// measurement_break_away_count
Ref: measurement_break_away_count.order_info - order_info.id
